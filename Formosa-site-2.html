<script>
/* util */
const BRL = n => (n||0).toLocaleString('pt-BR',{style:'currency',currency:'BRL'});
const WPP = "558981314000";
/* estado simples do carrinho
cada item do carrinho: { nome, total, linhas?: [{nome, unit, qty, sub}] } */
const cart = [];
function atualizarResumo(){
const total = cart.reduce((s,i)=>s+i.total,0);
document.getElementById('contagem').textContent = cart.length;
document.getElementById('totalCarrinho').textContent = BRL(total);
document.getElementById('cartFooterTotal').textContent = BRL(total);
document.getElementById('cartFooter').classList.toggle('show', total>0);
const host = document.getElementById('listaCarrinho');
if(!cart.length){ host.innerHTML = '<div class="muted">Seu carrinho está vazio.</div>';
return; }
host.innerHTML = cart.map((c,i)=>`
<div style="padding:10px 0; border-bottom:1px dashed #eee">
<div style="display:flex; justify-content:space-between; gap:12px; align-items:center;">
<div style="font-weight:600">${c.nome}</div>
<div style="display:flex; align-items:center; gap:8px">
<div style="white-space:nowrap; font-weight:700">${BRL(c.total)}</div>
<button class="btn danger" onclick="remover(${i})">Excluir</button>
</div>
</div>
${c.linhas ? `
<div style="margin-top:6px; padding-left:10px; color:#555; font-size:.92rem">
${c.linhas.map(L=>`<div>• ${L.nome} x${L.qty} • ${BRL(L.unit)} =
${BRL(L.sub)}</div>`).join('')}
</div>` : ``}
</div>
`).join('');
}
function adicionar(nome, total){
cart.push({nome,total});
atualizarResumo();
window.scrollTo({top:0,behavior:'smooth'});
}
function remover(i){
cart.splice(i,1);
atualizarResumo();
}
/* produto (editor simples) */
let totalAtual = 0;
let itensAtuais = []; // guarda itens atuais da cesta na tela
const itensBase = [
["Arroz 5kg", 24.90, 2],
["Feijão 2kg", 18.90, 1],
["Óleo 900ml", 6.49, 2],
["Açúcar 1kg", 4.50, 2],
["Café 250g", 8.90, 1],
["Macarrão 500g", 5.90, 2]
];
function verProduto(tipo){
const nome = tipo==='premium' ? "Cesta Básica Premium" :
tipo==='personalizada' ? "Cesta Personalizada" :
"Cesta Básica Tradicional";
document.getElementById('produtoNome').textContent = nome;
const host = document.getElementById('itens');
host.innerHTML = '';
itensAtuais = JSON.parse(JSON.stringify(itensBase));
if(tipo==='premium'){ itensAtuais.push(["Azeite 500ml",29.90,1]); }
if(tipo==='personalizada'){ itensAtuais.push(["Extras a escolher",0,0]); }
function linha(idx){
const [nome, preco] = itensAtuais[idx];
const q = itensAtuais[idx][2];
const subtotal = +(preco*q).toFixed(2);
return `
<div style="display:flex; justify-content:space-between; align-items:center; padding:8px
12px; border-bottom:1px dashed #eee">
<div>
<div style="font-weight:600">${nome}</div>
<div class="muted">${BRL(preco)}</div>
</div>
<div style="display:flex; align-items:center; gap:6px">
<button class="btn" onclick="alterar(${idx},-1)">–</button>
<div>${q}</div>
<button class="btn" onclick="alterar(${idx},1)">+</button>
<div style="width:88px; text-align:right; font-weight:700">${BRL(subtotal)}</div>
</div>
</div>`;
}
window.alterar = (i, d) => {
itensAtuais[i][2] = Math.max(0, itensAtuais[i][2] + d);
render();
};
function render(){
host.innerHTML = itensAtuais.map((_,i)=>linha(i)).join('');
totalAtual = +(itensAtuais.reduce((s,[,p,q])=>s+p*q,0)).toFixed(2);
document.getElementById('total').textContent = BRL(totalAtual);
}
render();
// navega para a seção produto
setActive('#produto');
}
/* Adiciona a cesta com suas linhas detalhadas ao carrinho */
function adicionarCesta(){
const nome = document.getElementById('produtoNome').textContent;
const linhas = itensAtuais
.filter(([ , , q]) => q>0)
.map(([n, unit, qty]) => ({ nome:n, unit, qty, sub: +(unit*qty).toFixed(2) }));
const total = linhas.reduce((s,l)=>s+l.sub,0);
cart.push({ nome, total, linhas });
atualizarResumo();
window.scrollTo({top:0,behavior:'smooth'});
}
/* navegação entre seções */
function setActiveByHash(){
const h = location.hash || '#home';
setActive(h);
}
function setActive(hash){
document.querySelectorAll('.pill').forEach(p=>p.classList.toggle('active',
p.getAttribute('href')===hash));
document.querySelectorAll('section').forEach(s=>s.classList.remove('active'));
const target = document.querySelector(hash);
(target || document.getElementById('home')).classList.add('active');
}
window.addEventListener('hashchange', setActiveByHash);
setActiveByHash();
/* MODAL de finalização */
function abrirModal(){
if(!cart.length){ alert('Carrinho vazio.'); return; }
document.getElementById('modal').classList.add('show');
}
function fecharModal(){ document.getElementById('modal').classList.remove('show'); }
/* finalizar via WhatsApp (coletando dados do modal) — com itens detalhados da cesta */
function finalizarWhats(){
const nome = document.getElementById('mNome').value.trim();
const whats = document.getElementById('mWhats').value.trim();
const rua = document.getElementById('mRua').value.trim();
const bairro = document.getElementById('mBairro').value.trim();
const cidade = document.getElementById('mCidade').value.trim();
const cep = document.getElementById('mCEP').value.trim();
const pagamento = document.getElementById('mPagamento').value;
const obs = document.getElementById('mObs').value.trim();
if(!nome || !whats){ alert('Preencha Nome e WhatsApp.'); return; }
const total = cart.reduce((s,i)=>s+i.total,0);
const linhas = [
'*Pedido — Formosa Qualidade*',
`Cliente: ${nome}`,
`WhatsApp: ${whats}`,
`Endereço: ${rua}, ${bairro}, ${cidade}, CEP ${cep}`,
`Forma de pagamento: ${pagamento}`,
'',
'*Itens:*'
];
cart.forEach(item=>{
if(item.linhas && item.linhas.length){
linhas.push(`- ${item.nome} — ${BRL(item.total)}`);
item.linhas.forEach(L=>{
linhas.push(` • ${L.nome} x${L.qty} • ${BRL(L.unit)} = ${BRL(L.sub)}`);
});
}else{
linhas.push(`- ${item.nome} — ${BRL(item.total)}`);
}
});
linhas.push('', `*Total: ${BRL(total)}*`);
if(obs){ linhas.push('', `Obs: ${obs}`); }
const msg = encodeURIComponent(linhas.join('\n'));
const url = `https://wa.me/${WPP}?text=${msg}`;
fecharModal();
window.open(url,'_blank');
}
/* init */
atualizarResumo();
